# 正则匹配

	let reg;
	reg.test(string) // true为匹配上了


### 1. 手机号码
	reg = /^((13[0-9])|(14[5,7])|(15[0-3,5-9])|(17[0,3,5-8])|(18[0-9])|166|198|199|147)\d{8}$/;

### 2. 电话号码
	reg = /^(\d{3,4}-)?(\d{7,14})+(-\d{1,4})?$/;

### 3. 数字
	reg = /^\d+(\.\d*)?$/;

### 4. 邮箱
	reg = /^\w+([-+.]\w+)*@\w+([-.]\w+)*\.\w+([-.]\w+)*$/;
